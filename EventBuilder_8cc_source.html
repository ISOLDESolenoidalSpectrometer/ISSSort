45

47 for( unsigned int j = 0; j < set->GetNumberOfArrayChannels(); ++j ) {
48

55

57

59 else if( j >= 11 && j <= 21 ) {
60
61 mystrip = j - 11;
62 array_nid[i].push_back( mystrip );
63 array_pid[i].push_back( -1 );

462 input_tree->DropBaskets();
463 if( i == 0 ) input_tree->GetEntry(idx);

1300 nptool_tree->GetEntry(i);
1301
1302 // ------------------------------------------- //
1303 // Find particles on the front strips of array //

1305 for( unsigned int j = 0; j < sim_data->GetMultiplicityFront(); j++ ){

1308 myrow = array_evt->FindRow( sim_data->GetFront_DetectorNbr(j) );

2714void ISSEventBuilder::CdFinder() {
2715

2717
2718 // Quick check that we have a CD at all
2719 if( set->GetNumberOfCDLayers() == 0 ) return;

2721 // Multiplicity, time and energy matrices
2722 for( unsigned int i = 0; i < set->GetNumberOfCDLayers(); ++i ) {
2723
2724 // Multiplicity
2725 if( cdren_list[i].size() || cdsen_list[i].size() )
2726 cd_rs_mult[i]->Fill( cdren_list[i].size(), cdsen_list[i].size() );
2727
2728 // Plot each sector energy against each ring energy
2729 // and ring/sector ID against energy
2730 for( unsigned int j = 0; j < cdsen_list[i].size(); j++ ) {
2731
2732 cd_s_en[i]->Fill( cdsid_list[i][j], cdsen_list[i][j]);

2737 cd_rs_td[i]->Fill( cdrtd_list[i][k] - cdstd_list[i][j] );
2738
2739 } // k - rings
2740
2741 } // j - sectors
2742
2743 // Do not double count ring ID vs energy events
2744 for( unsigned int k = 0; k < cdren_list[i].size(); k++ ) {
2745
2746 cd_r_en[i]->Fill( cdrid_list[i][k], cdren_list[i][k]);
2747 } // k - rings
2748
2749 } // i - layer
2750
2751 // Look for the dE signal as a trigger
2752 unsigned char dE_idx = set->GetCDEnergyLossStart();
2753
2754 // If we never found a sector in the dE at all, there's no "trigger"
2755 if( cdsen_list[dE_idx].size() == 0 ) return;
2756
2757 // If we have at least one sector and at least one ring, we have at least one event!
2758 if( cdsen_list[dE_idx].size() > 0 && cdren_list[dE_idx].size() > 0 ){
2759
2760 // Keep track of which events we've used
2761 std::vector<unsigned int> used_idx;
2762
2763 // Check all combinations for charge-sharing
2764 for( unsigned int j = 0; j < cdsen_list[dE_idx].size(); j++ ) {
2765
2766 // Check if it's been used already
2767 bool skip_flag = false;
2768 for( unsigned int xx = 0; xx < used_idx.size(); ++xx )
2769 if( used_idx[xx] == j ) skip_flag = true;
2770 if( skip_flag ) continue;
2771
2772 // Check all other sector hits to find a neighbour
2773 for( unsigned int k = 0; k < cdsen_list[dE_idx].size(); k++ ) {
2774
2775 // Not neighbours with each other
2776 if( j == k ) continue;
2777
2778 // Check if it's been used already
2779 skip_flag = false;
2780 for( unsigned int xx = 0; xx < used_idx.size(); ++xx )
2781 if( used_idx[xx] == k ) skip_flag = true;
2782 if( skip_flag ) continue;
2783
2784 // Check if they are really neighbours
2785 float sec_en_diff = cdsen_list[dE_idx][j] - cdsen_list[dE_idx][k];
2786 int sec_id_diff = cdsid_list[dE_idx][j] - cdsid_list[dE_idx][k];
2787 double sec_td_diff = cdstd_list[dE_idx][j] - cdstd_list[dE_idx][k];
2788 if( ( TMath::Abs( sec_id_diff ) == 1 ||
2789 TMath::Abs( sec_id_diff ) == set->GetNumberOfCDSectors()-1 ) &&
2790 TMath::Abs( sec_td_diff ) < set->GetCDRSHitWindow() ) {
2791
2792 // Charge-sharing add-back, use highest energy for sector ID and time
2793 float sumen = cdsen_list[dE_idx][j] + cdsen_list[dE_idx][k];
2794 int secid_tmp = cdsid_list[dE_idx][j];
2795 double sectd_tmp = cdstd_list[dE_idx][j];
2796 if( sec_en_diff < 0 ) {
2797 secid_tmp = cdsid_list[dE_idx][k];
2798 sectd_tmp = cdstd_list[dE_idx][k];
2799 }
2800
2801 // Mark both hits as used
2802 used_idx.push_back(j);
2803 used_idx.push_back(k);
2804
2805 // Make a particle event in the CD
2806 cd_evt.push_back( std::make_shared<ISSCDEvt>() );
2807
2808 // Add the first fragment and set the ring and sector
2809 cd_evt.back()->AddFragment( sumen, dE_idx );
2810 cd_evt.back()->SetSector( secid_tmp );
2811 cd_evt.back()->SetdETime( sectd_tmp );
2812
2813 // Finished with this, go back to the j loop
2814 skip_flag = true;
2815 break;
2816
2817 } // neighbours, do charge-sharing addback
2818
2819 } // all other sector hits - k
2820
2821 // Check if we used this in an add-back event
2822 if( skip_flag ) continue;
2823
2824 // We only get here if we didn't find a neighbour
2825 // So let's make a single event out of this
2826 cd_evt.push_back( std::make_shared<ISSCDEvt>() );
2827 cd_evt.back()->AddFragment( cdsen_list[dE_idx][j], dE_idx );
2828 cd_evt.back()->SetSector( cdsid_list[dE_idx][j] );
2829 cd_evt.back()->SetdETime( cdstd_list[dE_idx][j] );
2830
2831 // Mark hit as used before moving to next
2832 used_idx.push_back(j);
2833
2834 } // j
2835
2836 } // Multiple sectors and at least 1 ring
2837 // Now we've finished finding particles in the dE layer of the CD
2838
2839 // Find the matching rings for each particle based on time difference
2840 // We can re-use the same ring for each particle, because they are 360 and
2841 // we might be unlucky enough for both fragments to pileup in the same ring.
2842 for( unsigned int partid = 0; partid < cd_evt.size(); ++partid ) {
2843
2844 // Set the dE and log for finding the ring
2845 double sectime_cur = cd_evt[partid]->GetdETime();
2846
2847 // Find matching ring, easy if there is only 1
2848 int rindex_match = 0;
2849 double rs_tdiff_best = 2.0 * set->GetEventWindow();
2850
2851 // Search for the best
2852 // TODO: Maybe this should instead be energy difference?
2853 // TODO: Or if it's charge sharing, we need maximum energy?
2854 for( unsigned int j = 0; j < cdrtd_list[dE_idx].size(); ++j ) {
2855
2856 double rs_tdiff_cur = cdrtd_list[dE_idx][j] - sectime_cur;
2857 rs_tdiff_cur = TMath::Abs( rs_tdiff_cur );
2858 if( rs_tdiff_cur < rs_tdiff_best ) {
2859
2860 rindex_match = j;
2861 rs_tdiff_best = rs_tdiff_cur;
2862
2863 } // better match found
2864
2865 } // loop over all ring hits - j
2866
2867 // Set the ring for the dE layer
2868 if( rs_tdiff_best < set->GetCDRSHitWindow() )
2869 cd_evt[partid]->SetRing( cdrid_list[dE_idx][rindex_match] );
2870 else
2871 cd_evt[partid]->SetRing( 0xff ); // no ring found
2872
2873 } // partid - finished matching rings
2874
2875 // ------------------------------------------------------------------------------//
2876 // Now we need to loop over every other layer and find a match for each particle //
2877 // ------------------------------------------------------------------------------//
2878 for( unsigned int i = 0; i < set->GetNumberOfCDLayers(); ++i ) {
2879
2880 // Don't do the dE layer again though
2881 if( i == dE_idx ) continue;
2882
2883 // Mark up those that are used already
2884 std::vector<unsigned int> used_idx;
2885
2886 // Check every particle for a match in the other layers
2887 for( unsigned int partid = 0; partid < cd_evt.size(); ++partid ) {
2888
2889 // Find the matching sector in the available list
2890 for( unsigned int j = 0; j < cdsen_list[i].size(); ++j ) {
2891
2892 // Particles must be in the same sector
2893 if( cdsid_list[i][j] != cd_evt[partid]->GetSector() )
2894 continue;
2895
2896 // Check if it's been used already
2897 bool skip_flag = false;
2898 for( unsigned int xx = 0; xx < used_idx.size(); ++xx )
2899 if( used_idx[xx] == j ) skip_flag = true;
2900 if( skip_flag ) continue;
2901
2902 // Check all other sector hits to find a neighbour
2903 for( unsigned int k = 0; k < cdsen_list[i].size(); k++ ) {
2904
2905 // Not neighbours with each other
2906 if( j == k ) continue;
2907

2911 if( used_idx[xx] == k ) skip_flag = true;
2912 if( skip_flag ) continue;

2918 TMath::Abs( sec_id_diff ) == set->GetNumberOfCDSectors()-1 ) &&

2923

2925 used_idx.push_back(j);
2926 used_idx.push_back(k);
