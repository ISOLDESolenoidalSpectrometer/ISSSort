2690void ISSEventBuilder::CdFinder() {
2691
2692 //std::cout << __PRETTY_FUNCTION__ << std::endl;
2693
2694 // Multiplicity matrix
2695 for( unsigned int i = 0; i < set->GetNumberOfCDLayers(); ++i )
2696 if( cdren_list[i].size() || cdsen_list[i].size() )
2697 cd_rs_mult[i]->Fill( cdren_list[i].size(), cdsen_list[i].size() );
2698
2699 // Look for the dE signal as a trigger
2700 unsigned char dE_idx = set->GetCDEnergyLossStart();
2701
2702 // If we never found a sector in the dE at all, there's no "trigger"
2703 if( cdsen_list[dE_idx].size() == 0 ) return;
2704
2705 // If we have at least one sector and at least one ring, we have at least one event!
2706 if( cdsen_list[dE_idx].size() > 0 && cdren_list[dE_idx].size() > 0 ){
2707
2708 // Keep track of which events we've used
2709 std::vector<unsigned int> used_idx;
2710
2711 // Check all combinations for charge-sharing
2712 for( unsigned int j = 0; j < cdsen_list[dE_idx].size(); j++ ) {
2713
2714 // Check if it's been used already
2715 bool skip_flag = false;
2716 for( unsigned int xx = 0; xx < used_idx.size(); ++xx )
2717 if( used_idx[xx] == j ) skip_flag = true;
2718 if( skip_flag ) continue;
2719
2720 // Check all other sector hits to find a neighbour
2721 for( unsigned int k = 0; k < cdsen_list[dE_idx].size(); k++ ) {
2722
2723 // Not neighbours with each other
2724 if( j == k ) continue;
2725
2726 // Check if it's been used already
2727 skip_flag = false;
2728 for( unsigned int xx = 0; xx < used_idx.size(); ++xx )
2729 if( used_idx[xx] == k ) skip_flag = true;
2730 if( skip_flag ) continue;
2731
2732 // Check if they are really neighbours
2733 float sec_en_diff = cdsen_list[dE_idx][j] - cdsen_list[dE_idx][k];
2734 int sec_id_diff = cdsid_list[dE_idx][j] - cdsid_list[dE_idx][k];
2735 double sec_td_diff = cdstd_list[dE_idx][j] - cdstd_list[dE_idx][k];
2736 if( ( TMath::Abs( sec_id_diff ) == 1 ||
2737 TMath::Abs( sec_id_diff ) == set->GetNumberOfCDSectors()-1 ) &&
2738 TMath::Abs( sec_td_diff ) < set->GetCDRSHitWindow() ) {
2739
2740 // Charge-sharing add-back, use highest energy for sector ID and time
2741 float sumen = cdsen_list[dE_idx][j] + cdsen_list[dE_idx][k];
2742 int secid_tmp = cdsid_list[dE_idx][j];
2743 double sectd_tmp = cdstd_list[dE_idx][j];

2745 secid_tmp = cdsid_list[dE_idx][k];
2746 sectd_tmp = cdstd_list[dE_idx][k];
2747 }
2748
2749 // Mark both hits as used
2750 used_idx.push_back(j);
2751 used_idx.push_back(k);
2752
2753 // Make a particle event in the CD
2754 cd_evt.push_back( std::make_shared<ISSCDEvt>() );
2755

2757 cd_evt.back()->AddFragment( sumen, dE_idx );
2758 cd_evt.back()->SetSector( secid_tmp );
2759 cd_evt.back()->SetdETime( sectd_tmp );
2760
2761 // Finished with this, go back to the j loop
2762 skip_flag = true;
2763 break;
2764
2765 } // neighbours, do charge-sharing addback
2766
2767 } // all other sector hits - k
2768
2769 // Check if we used this in an add-back event
2770 if( skip_flag ) continue;
2771
2772 // We only get here if we didn't find a neighbour
2773 // So let's make a single event out of this
2774 cd_evt.push_back( std::make_shared<ISSCDEvt>() );
2775 cd_evt.back()->AddFragment( cdsen_list[dE_idx][j], dE_idx );
2776 cd_evt.back()->SetSector( cdsid_list[dE_idx][j] );
2777 cd_evt.back()->SetdETime( cdstd_list[dE_idx][j] );
2778
2779 // Mark hit as used before moving to next
2780 used_idx.push_back(j);
2781
2782 } // j
2783
2784 } // Multiple sectors and at least 1 ring

2786
2787 // Find the matching rings for each particle based on time difference
2788 // We can re-use the same ring for each particle, because they are 360˚ and
2789 // we might be unlucky enough for both fragments to pileup in the same ring.
2790 for( unsigned int partid = 0; partid < cd_evt.size(); ++partid ) {
2791
2792 // Set the dE and log for finding the ring
2793 double sectime_cur = cd_evt[partid]->GetdETime();
2794
2795 // Find matching ring, easy if there is only 1
2796 int rindex_match = 0;
2797 double rs_tdiff_best = 2.0 * set->GetEventWindow();
2798
2799 // Search for the best
2800 // TODO: Maybe this should instead be energy difference?
2801 // TODO: Or if it's charge sharing, we need maximum energy?

2803

2806 if( rs_tdiff_cur < rs_tdiff_best ) {
2807
2808 rindex_match = j;
2809 rs_tdiff_best = rs_tdiff_cur;
2810
2811 } // better match found
2812
2813 } // loop over all ring hits - j
2814
2815 // Set the ring for the dE layer
2816 if( rs_tdiff_best < set->GetCDRSHitWindow() )
2817 cd_evt[partid]->SetRing( cdrid_list[dE_idx][rindex_match] );
2818 else
2819 cd_evt[partid]->SetRing( 0xff ); // no ring found
2820
2821 } // partid - finished matching rings
2822
2823 // ------------------------------------------------------------------------------//
2824 // Now we need to loop over every other layer and find a match for each particle //
2825 // ------------------------------------------------------------------------------//
2826 for( unsigned int i = 0; i < set->GetNumberOfCDLayers(); ++i ) {
2827
2828 // Don't do the dE layer again though
2829 if( i == dE_idx ) continue;
2830
2831 // Mark up those that are used already
2832 std::vector<unsigned int> used_idx;
2833
2834 // Check every particle for a match in the other layers
2835 for( unsigned int partid = 0; partid < cd_evt.size(); ++partid ) {
2836
2837 // Find the matching sector in the available list
2838 for( unsigned int j = 0; j < cdsen_list[i].size(); ++j ) {
2839
2840 // Particles must be in the same sector
2841 if( cdsid_list[i][j] != cd_evt[partid]->GetSector() )
2842 continue;
2843
2844 // Check if it's been used already
2845 bool skip_flag = false;
2846 for( unsigned int xx = 0; xx < used_idx.size(); ++xx )
2847 if( used_idx[xx] == j ) skip_flag = true;
2848 if( skip_flag ) continue;
2849
2850 // Check all other sector hits to find a neighbour
2851 for( unsigned int k = 0; k < cdsen_list[i].size(); k++ ) {
2852
2853 // Not neighbours with each other
2854 if( j == k ) continue;
2855
2856 // Check if it's been used already
2857 skip_flag = false;
2858 for( unsigned int xx = 0; xx < used_idx.size(); ++xx )
2859 if( used_idx[xx] == k ) skip_flag = true;
2860 if( skip_flag ) continue;
2861
2862 // Check if they are really neighbours
2863 int sec_id_diff = cdsid_list[i][j] - cdsid_list[i][k];
2864 double sec_td_diff = cdstd_list[i][j] - cdstd_list[i][k];
2865 if( ( TMath::Abs( sec_id_diff ) == 1 ||
2866 TMath::Abs( sec_id_diff ) == set->GetNumberOfCDSectors()-1 ) &&
2867 TMath::Abs( sec_td_diff ) < set->GetCDRSHitWindow() ) {
2868
2869 // Charge-sharing add-back
2870 float sumen = cdsen_list[i][j] + cdsen_list[i][k];
2871
2872 // Mark both hits as used
2873 used_idx.push_back(j);
2874 used_idx.push_back(k);
2875
2876 // Add fragment for this particle and set the E time if needed
2877 cd_evt[partid]->AddFragment( sumen, i );
2878 if( i == set->GetCDEnergyRestStart() )
2879 cd_evt[partid]->SetETime( cdstd_list[i][j] );
2880
2881 // Finished with this, go back to the j loop
2882 skip_flag = true;
2883 break;
2884
2885 } // neighbours, do charge-sharing addback
2886
2887 } // all other sector hits - k
2888
2889 // Check if we did charge-sharing
2890 if( skip_flag ) continue;
2891
2892 // Add fragment for event 1 and set the E time if needed
2893 cd_evt[partid]->AddFragment( cdsen_list[i][j], i );
2894 if( i == set->GetCDEnergyRestStart() )
2895 cd_evt[partid]->SetETime( cdstd_list[i][j] );
2896
2897 // Mark hit as used and move to the next particle
2898 used_idx.push_back(j);

2900

2913 cd_evt[partid]->GetEnergyLoss( set->GetCDEnergyLossStart(), set->GetCDEnergyLossStop() ) );
