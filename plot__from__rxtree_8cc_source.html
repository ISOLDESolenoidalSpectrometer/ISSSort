29 // Make a reaction instance, etc

33 // Set branches
34 ISSRxEvent *myrxevt = new ISSRxEvent();
35 ISSRxInfo *myrxinfo = new ISSRxInfo();
36 t->SetBranchAddress( "RxEvent", &myrxevt );

39 // Make a histogram or two
40 TH2F *zphimap = new TH2F( "zphimap", "z-phi hit map", 1000, -1000, 1000, 560, -100, 460 );
41 TH2F *pmodmap = new TH2F( "pmodmap", "p vs module hit map", 512, -0.5, 511.5, 3, -0.5, 2.5 );
42 TH2F *pnmap = new TH2F( "pnmap", "p vs n hit map", 512, -0.5, 511.5, 264, -0.5, 263.5 );
43
44 // Loop over events
45 for( unsigned int i = 0; i < t->GetEntries(); i++ ){
46
47 // Get entry

49
50 // Calculate module number

52 double phi = myrxevt->GetPhi();
53 int mymod = myrxevt->GetModule();

58 // Plot hit maps

60 pmodmap->Fill( myrow*128 + mypid, mymod );

63 }
