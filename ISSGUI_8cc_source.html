415void ISSGUI::on_del_clicked() {
416
417 // Cleanup the selected entries
418 run_selected->Clear();
419
420 // Slot to react to remove file button
421 if( run_list_box->GetSelected() < 0 ) {
422
423 std::cout << "\nNo runs selected" << std::endl;
424
425 }
426
427 else if( run_list_box->GetMultipleSelections() ) {
428
429 run_list_box->GetSelectedEntries( run_selected );
430 TIter next( run_selected );
431
432 while( TGLBEntry *ent = (TGLBEntry*)next() ) {
433
434 filestatus.at( ent->EntryId() - 1 ) = false;
435 run_list_box->RemoveEntry( ent->EntryId() );
436
437 }
438
439 }
440
441 else {
442
443 filestatus.at( run_list_box->GetSelected() - 1 ) = false;
444 run_list_box->RemoveEntry( -1 );
445
446 }
447
448 run_list_box->Layout();
449

452void ISSGUI::SaveSetup( TString setupfile ) {
453
454 TEnv *fSetup = new TEnv( setupfile );
455
456 TString list_of_files = "";

481void ISSGUI::LoadSetup( TString setupfile ) {
482
483 TEnv *fSetup = new TEnv( setupfile );
484 fSetup->ReadFile( setupfile, kEnvLocal );
485
486 TString list_of_files = fSetup->GetValue( "filelist", "" );
487 std::vector <TString>().swap( filelist );
488 std::vector <bool>().swap( filestatus );
489 run_list_box->RemoveAll();
490
491 std::stringstream ss;
492 TString current_file;
493 ss << list_of_files;
494 ss >> current_file;
495
496 while( !ss.eof() ) {
497
498 filelist.push_back( current_file );
499 filestatus.push_back( true );
500
501 run_list_box->AddEntrySort( current_file, filelist.size() );
502
503 ss >> current_file;
504
505 }
506
507 run_list_box->Layout();
508
509 check_source->SetOn( fSetup->GetValue( "source", false ) );
510 check_force->SetOn( fSetup->GetValue( "force", false ) );
511 check_event->SetOn( fSetup->GetValue( "event", false ) );
512
513 text_set_file->SetText( fSetup->GetValue( "settings", "dummy" ) );
514 text_cal_file->SetText( fSetup->GetValue( "calibration", "dummy" ) );
515 text_rea_file->SetText( fSetup->GetValue( "reaction", "dummy" ) );
516 text_out_file->SetText( fSetup->GetValue( "output", "output.root" ) );
517
518}

520void ISSGUI::gui_convert(){
521
522 //------------------------//
523 // Run conversion to ROOT //
524 //------------------------//
525 ISSConverter conv( myset );
526 conv.AddCalibration( mycal );
527 conv.AddProgressBar( prog_conv );
528 std::cout << "\n +++ ISS Analysis:: processing Converter +++" << std::endl;
529
530 // Progress bar and filename
531 std::string prog_format;
532 TFile *rtest;
533 std::ifstream ftest;
534 TString name_input_file;
535 TString name_output_file;
536
537 // Check each file
538 for( unsigned int i = 0; i < filelist.size(); i++ ){
539
540 name_input_file = filelist.at(i);
541 if( flag_source ) name_output_file = filelist.at(i) + "_source.root";
542 else name_output_file = filelist.at(i) + ".root";
543
544 force_convert.push_back( false );
545
546 // Skip the file if it's deleted
547 if( !filestatus.at(i) ) continue;
548
549 // If input doesn't exist, skip it
550 ftest.open( name_input_file.Data() );
551 if( !ftest.is_open() ) {
552
553 std::cerr << name_input_file << " does not exist" << std::endl;
554 continue;
555
556 }
557 else ftest.close();
558
559 // If output doesn't exist, we have to convert it anyway
560 // The convert flag will force it to be converted
561 ftest.open( name_output_file );
562 if( !ftest.is_open() ) force_convert.at(i) = true;
563 else {
564
565 ftest.close();
566 rtest = new TFile( name_output_file );
567 if( rtest->IsZombie() ) force_convert.at(i) = true;
568 if( !flag_convert && !force_convert.at(i) )
569 std::cout << name_output_file << " already converted" << std::endl;
570 rtest->Close();
571
572 }
573
574 if( flag_convert || force_convert.at(i) ) {
575
576 std::cout << name_input_file << " --> ";
577 std::cout << name_output_file << std::endl;
578
579 prog_format = "Converting ";
580 prog_format += name_input_file( name_input_file.Last('/') + 1,
581 name_input_file.Length() - name_input_file.Last('/') ).Data();
582 prog_format += ": %.0f%%";
583 prog_conv->ShowPosition( true, false, prog_format.data() );
584
585 conv.SetOutput( name_output_file.Data() );
586 if( flag_source ) conv.SourceOnly();
587 conv.MakeTree();
588 conv.MakeHists();
589 conv.ConvertFile( name_input_file.Data() );
590
591 prog_format = "Converter complete";
592 prog_conv->ShowPosition( true, false, prog_format.data() );
593
594 // Time sorting
595 if( !flag_source ) {
596
597 prog_format = "Time ordering ";
598 prog_format += name_input_file( name_input_file.Last('/') + 1,
