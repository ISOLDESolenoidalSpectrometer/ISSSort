33void z_to_angle_simulator( std::string reaction_file ){
34
35 // Make a reaction instance
36 myset = std::make_shared<ISSSettings>( "dummy" );
37 myreact = std::make_shared<ISSReaction>( reaction_file, myset, false );
38
39 // Array distance
40 double z_0 = myreact->GetArrayDistance();
41
42 // A graph of Ex versus ThetaCM for this z value
43 g = new TGraph2D();
44 g->SetTitle("ThetaCM as a function of z_meas and Ex;z_meas [mm];E_{x} [keV]");
45
46 // Do the reaction for a range of proton energies
47 for( unsigned int i = 0; i < ncalcs+1; ++i ){
48
49 // Calculate the proton energy
50 double en = min_energy + ( max_energy - min_energy ) / ncalcs * (double)i;
51
52 // Do the reaction for a range of z positions
53 for( unsigned int j = 0; j < ncalcs+1; ++j ){
54
55 // Calculate the z position
56 double z_det = min_z + ( max_z - min_z ) / ncalcs * (double)j;
57 double z_meas = z_det + z_0; // forward
58 if( z_0 < 0 ) z_meas = z_0 - z_det; // backwards
59
60 // Make a vector for the interaction position
61 TVector3 pos( radius, 0.0, z_det );
62
63 // Make the reaction (this should be actually the simulation function)
64 myreact->MakeReaction( pos, en );
65
66 std::cout << en << "\t" << z_det << "\t";
67 std::cout << z_0 << "\t" << z_meas << "\t";
68 std::cout << myreact->GetDistance() << "\t";
69 std::cout << myreact->GetEx() << "\t";
70 std::cout << myreact->GetThetaCM() << std::endl;
71
72 // Common sense checks
73 if( TMath::IsNaN( myreact->GetEx() ) || TMath::IsNaN( myreact->GetThetaCM() ) ) continue;
74 if( myreact->GetEx() < -100 || myreact->GetEx() > 1e6 ) continue;
75
76 // Fill graph
77 g->SetPoint( g->GetN(), z_meas, myreact->GetEx(), myreact->GetThetaCM() );
78
79 }
