510 if( cd_etot_start >= n_cd_layer && n_cd_layer != 0 ){
511 std::cerr << "CDEnergyTotalStart must be less than NumberOfRecoilLayers" << std::endl;
512 std::cerr << "Reverting CDEnergyTotalStart to default value of 0" << std::endl;
513 cd_etot_start = 0;
514 }
515 if( cd_etot_stop >= n_cd_layer && n_cd_layer != 0 ){
516 std::cerr << "CDEnergyTotalStop must be less than NumberOfRecoilLayers" << std::endl;
517 std::cerr << "Reverting CDEnergyTotalStop to default value of 1" << std::endl;
518 cd_etot_stop = 1;
519 }
520 cd_vme.resize( n_cd_layer );
521 cd_mod.resize( n_cd_layer );
522 cd_ch.resize( n_cd_layer );
523 cd_layer.resize( GetNumberOfVmeCrates() );
524 cd_strip.resize( GetNumberOfVmeCrates() );
525 cd_side.resize( GetNumberOfVmeCrates() );
526
527 for( unsigned int i = 0; i < GetNumberOfVmeCrates(); ++i ){
528
529 cd_layer[i].resize( GetMaximumNumberOfVmeModules() );
530 cd_strip[i].resize( GetMaximumNumberOfVmeModules() );
531 cd_side[i].resize( GetMaximumNumberOfVmeModules() );
532
533 for( unsigned int j = 0; j < GetMaximumNumberOfVmeModules(); ++j ){
534
535 for( unsigned int k = 0; k < GetMaximumNumberOfVmeChannels(); ++k ){
536
537 cd_layer[i][j].push_back( -1 );
538 cd_strip[i][j].push_back( -1 );
539 cd_side[i][j].push_back( -1 );
540
541 }
542
543 }
544
545 }
546
547 for( unsigned int i = 0; i < n_cd_layer; ++i ){
548
549 cd_vme[i].resize( n_cd_side );
550 cd_mod[i].resize( n_cd_side );
551 cd_ch[i].resize( n_cd_side );
552
553 for( unsigned int j = 0; j < n_cd_side; ++j ){
554
555 std::string sidechar = "R";
556 unsigned int nstrips = n_cd_ring;
557 if( j == 1 ) {
558 sidechar = "S";
559 nstrips = n_cd_sector;
560 }
561
562 cd_vme[i][j].resize( nstrips );
563 cd_mod[i][j].resize( nstrips );
564 cd_ch[i][j].resize( nstrips );
565
566 for( unsigned int k = 0; k < nstrips; ++k ){
567
568 unsigned char mm = 2*i;
569 unsigned char cc = k;
570 if( k > GetMaximumNumberOfVmeChannels() ) {
571 mm++;
572 cc -= GetMaximumNumberOfVmeChannels();
573 }
574 cd_vme[i][j][k] = (int)config->GetValue( Form( "CD_%d_%d.%s.Crate", i, j, sidechar.data() ), (int)1 ); // layer, ring
575 cd_mod[i][j][k] = (int)config->GetValue( Form( "CD_%d_%d.%s.Module", i, j, sidechar.data() ), (int)mm );
576 cd_ch[i][j][k] = (int)config->GetValue( Form( "CD_%d_%d.%s.Channel", i, j, sidechar.data() ), (int)cc );
577
578 if( cd_vme[i][j][k] < GetNumberOfVmeCrates() &&
579 cd_mod[i][j][k] < GetMaximumNumberOfVmeModules() &&
580 cd_ch[i][j][k] < GetMaximumNumberOfVmeChannels() ) {
581
582 cd_strip[cd_vme[i][j][k]][cd_mod[i][j][k]][cd_ch[i][j][k]] = k;
583 cd_side[cd_vme[i][j][k]][cd_mod[i][j][k]][cd_ch[i][j][k]] = j;
584 cd_layer[cd_vme[i][j][k]][cd_mod[i][j][k]][cd_ch[i][j][k]] = i;
585
586 }
587
588 else {
589
590 std::cerr << "Dodgy CD settings:";
591 std::cerr << " crate = " << (int)cd_vme[i][j][k];
592 std::cerr << " module = " << (int)cd_mod[i][j][k];
593 std::cerr << " channel = " << (int)cd_ch[i][j][k] << std::endl;
594
595 }
596
597 }
598
599 }
600
601 }
602
603
604 // Perform a couple of sanity checks
605 if( n_caen_mod > 16 ) {
606
607 n_caen_mod = 16;
608 std::cout << "Maximum number of CAEN modules is 16" << std::endl;
609
610 }
611
612 if( n_mesy_mod > 16 ) {
613
614 n_mesy_mod = 16;
615 std::cout << "Maximum number of Mesytec modules is 16" << std::endl;
616
617 }
618
619 // Finished
620 delete config;
621
622}
