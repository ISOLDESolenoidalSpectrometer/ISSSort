59double butler_function( double *x, double *params ){
60
61 // Equation to solve for z, LHS = 0
62 double z = x[0];
63 double z_meas = params[0];
64 double r_meas = params[1];
65 double p = params[2]; // p
66 double qb = params[3]; // qb / 2*pi
67
68 // From Sam Bennett's first derivation, modified by A. Ceulemans
69 double theta_lab = TMath::ACos( qb * z / p );
70 double r_max = TMath::Abs( 2.0 * p * TMath::Sin( theta_lab ) / ( qb * TMath::TwoPi()) );
71 double psi = 2 * TMath::ASin( r_meas / r_max );
72 double root = z_meas - z * ( 1.0 - psi / TMath::TwoPi() );
73
74 return root;
75

96double theta_cm_function( double *x, double *params ){
97
98 // Input to the function
99 double theta_cm = x[0];
100 double z_meas = params[0];
101 double r_meas = params[1];
102 double Ex = params[2];
103 double qb = params[3]; // over 2*pi
104 double T1 = params[4];
105 double m1 = params[5];
106 double m2 = params[6];
107 double m3 = params[7];
108 double m4 = params[8];
109
110 // Invariant mass = total energy in centre of mass frame
111 double Etot_cm = m1*m1 + m2*m2;
112 Etot_cm += 2.0 * ( T1 + m1 ) * m2;
113 Etot_cm = TMath::Sqrt( Etot_cm );
114
115 // Work out what gamma and beta are
116 double E_lab = T1 + m1 + m2;
117 double gamma = E_lab / Etot_cm;
118 double beta = TMath::Sqrt( 1.0 - 1.0 / TMath::Power( gamma, 2.0 ) );
119
120 // What about the energy of the ejectile after the collision?
121 double e3_cm = TMath::Power( Etot_cm, 2.0 );
122 e3_cm -= TMath::Power( m4 + Ex, 2.0 );
123 e3_cm += TMath::Power( m3, 2.0 );
124 e3_cm /= 2.0 * Etot_cm;
125
126 // Now the momentum and velocity of the ejectile
127 double p3_cm = TMath::Power( e3_cm, 2.0 );
128 p3_cm -= TMath::Power( m3, 2.0 );
129 p3_cm = TMath::Sqrt( p3_cm );
130
131 // From Daniel Clarke:
132 //double p_lab = beta * E_lab;
133 double theta_lab = p3_cm * TMath::Sin( TMath::Pi() - theta_cm );
134 theta_lab /= p3_cm * TMath::Cos( TMath::Pi() - theta_cm ) + beta * e3_cm;
135 theta_lab /= gamma;
136 theta_lab = TMath::ATan( theta_lab );
137 if( theta_lab < 0 ) theta_lab = TMath::Pi() + theta_lab;
138
139 // Energy of ejectile in the lab
140 double e3_lab = gamma * beta * p3_cm * TMath::Cos( TMath::Pi() - theta_cm );
141 e3_lab += gamma * e3_cm;
142 double p3_lab = TMath::Sqrt( e3_lab*e3_lab - m3*m3 );
143
144 // Maximum radius of particle, z position and missing orbit fraction (psi)
145 double r_max = TMath::Abs( 2.0 * p3_lab * TMath::Sin( theta_lab ) / ( qb * TMath::TwoPi()) );
146 double z = p3_lab * TMath::Cos( theta_lab ) / qb;
147 double psi = 2.0 * TMath::ASin( r_meas / r_max );
148
149 // This is the equation to find the root of
150 double root = z_meas - z * ( 1.0 - psi / TMath::TwoPi() );
151
152 return root;
153
154}

386int ISSReaction::ReadCoordinates( std::string line, double *x, double *y ) {
387
388 int n_conversions;
389 int index = -1;
390 double x_temp, y_temp;
391
392 // Remove any comments beginning with #.
393 line.erase( std::find( line.begin(), line.end(), '#' ), line.end() );
394
395 // Check if line is empty (including whitespace).
396 int n = sscanf(line.c_str(), " %n", &index);
397 if( n == 0 && /* No successful conversions. */
398 index != -1 && /* %n was reached, so index no longer -1. */
399 line[index] == '\0') /* String actually ends at the format string end. */
400 return 0; // Nothing to convert.
401
402 index = -1;
403 n_conversions = sscanf(line.c_str(), "%lf %lf %n", &x_temp, &y_temp, &index);
404 if( n_conversions == 2 && /* Two successful conversions. */
405 index !=-1 && /* %n was reached, so index no longer -1. */
406 line[index] == '\0'){ /* String actually ends at the format string end. */
407
408 *x = x_temp; // Only store the x y pair if conversion was successful.
409 *y = y_temp;
410 return 1; // Successfully converted two doubles.
411
412 }
413
414 return 2; // Failed to convert string.
415
416}

423
424 std::shared_ptr<TCutG> cut;
425 std::vector<double> x_vector, y_vector;
426 int counter = 0;
427 std::string line;
428
429 // Read file line by line.
430 while( getline(*cut_file,line) ) {
431
432 int converted;
433 double x,y;
434 converted = ReadCoordinates(line, &x, &y);
435 if( !converted ) continue; // Line is empty (or is a comment).
436

440 return std::make_shared<TCutG>();
441 }
442
443 x_vector.push_back(x);
444 y_vector.push_back(y);
445 counter++;
446
447 }
448

450 cut->SetVarX( "x" );
451 cut->SetVarY( "y" );
452 for( int i = 0; i < counter; i++ )
453 cut->SetPoint( i, x_vector[i], y_vector[i] );

466 std::ifstream input_file;
467 input_file.open( AME_FILE );
468
469 std::string line, BE_str, N_str, Z_str;
470 std::string startline = "1N-Z";
471
472 short Ai, Zi, Ni;
473
474 // Loop over the file
475 if( input_file.is_open() ){
476
477 // Read first line
478 std::getline( input_file, line );
479
480 // Look for start of data
481 while( line.substr( 0, startline.size() ) != startline ){
482

488 exit(1);

491
492 }
493
494 // Read one more nonsense line with the units on
495 std::getline( input_file, line );
496
497 // Now process the data
498 while( std::getline( input_file, line ) ){
499
500 // Get mass excess from the line

507 Zi = std::stoi( Z_str );

942double ISSReaction::GetEnergyLoss( double Ei, double dist, std::unique_ptr<TGraph> &g ) {
943
944 unsigned int Nmeshpoints = 50; // number of steps to take in integration
945 double dx = dist/(double)Nmeshpoints;
946 double E = Ei;
947
948 // Create a spline for the provided TGraph
949 // Use akima spline to make energy loss graph smoother
950 std::unique_ptr<TSpline3> spline = std::make_unique<TSpline3>("spline", g.get(), "akima");
951 for( unsigned int i = 0; i < Nmeshpoints; i++ ) {
952
953 double eloss = spline->Eval(E) * dx;
954 if( eloss > E && eloss > 0 ) eloss = E; // incase we are "stopped"
955
956 E -= eloss;
957 if( E < 100. && eloss > 0 ) break; // when we fall below 100 keV we assume maximum energy loss
958
959 }
960
961 return Ei - E;
962
963}

975double ISSReaction::GetNuclearEnergyLoss( double Ei, double range, std::unique_ptr<TGraph> &gn, std::unique_ptr<TGraph> &gtot ) {
976
977 unsigned int Nmeshpoints = 200; // number of steps to take in integration
978 double dx = range/(double)Nmeshpoints;
979 double E = Ei;
980 double En = 0;
981
982 // Create splines for the provided TGraphs
983 std::unique_ptr<TSpline3> splineTot = std::make_unique<TSpline3>("splineTot", gtot.get(), "akima"); // Use akima spline to make energy loss graph smoother - BRJ
984 std::unique_ptr<TSpline3> splineN = std::make_unique<TSpline3>("splineN", gn.get(), "akima"); // Use akima spline to make energy loss graph smoother - BRJ
985
986 for( unsigned int i = 0; i < Nmeshpoints; i++ ) {
987 if( E < 0.5 ) break; // when we fall below 0.5 keV we assume we're stopped
988 E -= splineTot->Eval(E) * dx;
989 En += splineN->Eval(E) * dx;
990 }
991
992 return En;
993
