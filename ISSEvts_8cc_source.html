322TVector2 ISSArrayEvt::GetPhiXY(){
323
331
332 // Start with a flat wafer pointing to the sky
333 float x = array_radius; // diameter is 54.0 mm according to Chris Everett's CAD drawings
334 float y = nid%11 - 5.0; // take strip relative to centre (11 strips per face: 22 strips per module)
335 y *= nstrip_pitch; // n-side strip pitch = 2.0 mm

401int ISSArrayEvt::FindModule( double phi ){
402
404
405 // First rotate us so we have edge of module 0 vertical, +60
406 phi += TMath::Pi() / 3.;
407
408 // Then put us in the 0 -> 2 range
409 if( phi < 0.0 ) phi += TMath::TwoPi();
410 if( phi > TMath::TwoPi() ) phi -= TMath::TwoPi();

417int ISSArrayEvt::FindRow( double z ){
418
420
421 // Loop over each row
422 for( unsigned char i = 0; i < 4; i++ ){
423
424 // z0 is defined to edge of first wafer, so we start from 0
425 if( z > (double)(3-i) * wafer_length + (double)(3-i) * wafer_gap &&
426 z <= (double)(4-i) * wafer_length + (double)(3-i) * wafer_gap )

436int ISSArrayEvt::FindPID( double z ){
437
439
440 // First get the row number
441 char row = FindRow( z );
442
443 // Straight away return 0 if we don't hit the silicon
444 if( row < 0 ) return -1;
445
446 // Physical silicon length and gaps
447
448 // Shift the z to within a single silicon
449 z -= (double)(3-row) * ( wafer_length + wafer_gap );
450
451 // Shift the z to edge of the first strip
452 z -= wafer_guard;
453
454 // Loop over each strip
455 for( unsigned char i = 0; i < 128; i++ ){
456
457 // z0 is defined to edge of first wafer, so we start from 0
458 if( z > (double)(127-i) * pstrip_pitch &&
459 z <= (double)(128-i) * pstrip_pitch )
460 return i;
461
462 }
463
464 // everything else won't hit the silicon
465 return -1;
466

469int ISSArrayEvt::FindNID( double phi ){
470
472
473 // First get the module number
474 char mod = FindModule( phi );
475
476 // Straight away return 0 if we don't hit the silicon
477 if( mod < 0 ) return -1;
478
479 // Shift the phi to within a single module
480 phi -= (double)mod * TMath::TwoPi() / 3.;
481
482 // Rotate module 0 edge to vertical
483 phi += TMath::TwoPi() / 6.;
484
485 // Then put us in the 0 -> 2 range
486 if( phi < 0.0 ) phi += TMath::TwoPi();
487 if( phi > TMath::TwoPi() ) phi -= TMath::TwoPi();
488
489 // Check if we are on side A or B
490 bool sideB = false;
491 if( phi > TMath::TwoPi() / 6. ) {
492
493 sideB = true;
494 phi -= TMath::TwoPi() / 6.;
495
496 }
497
498 // Loop over each strip
499 for( unsigned char i = 0; i < 11; i++ ){
500
501 // Vector for the strip edge
502 TVector2 vec_low( array_radius, ( (double)i-5.5 ) * nstrip_pitch );
503 TVector2 vec_upp( array_radius, ( (double)i-4.5 ) * nstrip_pitch );
504
505 // Rotate by pi/6 to get it aligned with reference, i.e. +30
506 // this puts the edge of the wafer in the vertical position
507 vec_low = vec_low.Rotate( TMath::Pi() / 6. );
508 vec_upp = vec_upp.Rotate( TMath::Pi() / 6. );
509
510 // check if we are in the strip
511 if( phi > vec_low.Phi() && phi < vec_upp.Phi() )
512 return i + 11 * sideB;
513
514 }
515
516 // everything else won't hit the silicon
517 return -1;
518
519}

661TVector2 ISSCDEvt::GetPhiXY( bool randomised ){
662
670
671 // Get the ring and sector ids
672 float ringpos = (float)ring + 0.5;
673 float secpos = (float)sec + 0.5;
674
675 // and check if we need to randomise them
676 if( randomised ) {
677
678 ringpos = (float)ring + gRandom->Rndm();
679 secpos = (float)sec + gRandom->Rndm();
680
681 }
682
683 // Start with a sector pointing to the sky
684 double x = cd_inner_radius + ringpos * cd_ring_width + ring * cd_ring_gap;
685 TVector2 vec( x, 0 );
686
687 // This vector can now be rotated to the correct position
688 double phi = cd_phi_offset + secpos * cd_sector_pitch + sec * cd_sector_gap;
689 phi *= TMath::DegToRad();
690 vec = vec.Rotate( phi );
691
692 return vec;
693
