231float ISSArrayEvt::GetZ(){
232
240
241 float d = 127.5 - (float)pid; // take centre of the end strip
242 d *= pstrip_pitch; // p-side strip pitch = 0.953 mm
243 d += wafer_guard; // distance from wafer edge to active region
244 d += ( wafer_length + wafer_gap ) * (3.0 - (float)row); // move to correct row
245
246 return d; // in mm
247

250TVector2 ISSArrayEvt::GetPhiXY(){
251
259
260 // Start with a flat wafer pointing to the sky
261 float x = array_radius; // diameter is 54.0 mm according to Chris Everett's CAD drawings
262 float y = nid%11 - 5.0; // take strip relative to centre (11 strips per face: 22 strips per module)
263 y *= nstrip_pitch; // n-side strip pitch = 2.0 mm
264
265 // This vector can now be rotated to the correct position
266 TVector2 vec( x, y );
267
268 vec = vec.Rotate( -1.0 * TMath::Pi() / 6. ); // first face is at -30˚
269 vec = vec.Rotate( mod * TMath::TwoPi() / 3. ); // rotate 120˚ for each module
270 if( nid > 10 )
271 vec = vec.Rotate( TMath::TwoPi() / 6. ); // rotate 60˚ for each face
272
273 return vec;

329int ISSArrayEvt::FindModule( double phi ){
330
332
333 // First rotate us so we have edge of module 0 vertical, +60˚
334 phi += TMath::Pi() / 3.;
335
336 // Then put us in the 0 -> 2π range
337 if( phi < 0.0 ) phi += TMath::TwoPi();
338 if( phi > TMath::TwoPi() ) phi -= TMath::TwoPi();
339
340 // Return value is integer of 2π/3
341 return (int)( phi / ( TMath::TwoPi() / 3. ) );
342
343}

345int ISSArrayEvt::FindRow( double z ){
346
348
349 // Loop over each row
350 for( unsigned char i = 0; i < 4; i++ ){
351
352 // z0 is defined to edge of first wafer, so we start from 0
353 if( z > (double)(3-i) * wafer_length + (double)(3-i) * wafer_gap &&
354 z <= (double)(4-i) * wafer_length + (double)(3-i) * wafer_gap )
355 return i;
356
357 }
358
359 // everything else won't hit the silicon
360 return -1;
361
362}

364int ISSArrayEvt::FindPID( double z ){
365
367
368 // First get the row number
369 char row = FindRow( z );
370
371 // Straight away return 0 if we don't hit the silicon
372 if( row < 0 ) return -1;
373
374 // Physical silicon length and gaps
375
376 // Shift the z to within a single silicon
377 z -= (double)(3-row) * ( wafer_length + wafer_gap );
378
379 // Shift the z to edge of the first strip
380 z -= wafer_guard;
381
382 // Loop over each strip
383 for( unsigned char i = 0; i < 128; i++ ){
384
385 // z0 is defined to edge of first wafer, so we start from 0
386 if( z > (double)(127-i) * pstrip_pitch &&
387 z <= (double)(128-i) * pstrip_pitch )
388 return i;
389
390 }
391
392 // everything else won't hit the silicon
393 return -1;
394
395}

397int ISSArrayEvt::FindNID( double phi ){
398
400
401 // First get the module number
402 char mod = FindModule( phi );
403
404 // Straight away return 0 if we don't hit the silicon
405 if( mod < 0 ) return -1;
406
407 // Shift the phi to within a single module
408 phi -= (double)mod * TMath::TwoPi() / 3.;
409
410 // Rotate module 0 edge to vertical
411 phi += TMath::TwoPi() / 6.;
412
413 // Then put us in the 0 -> 2π range
414 if( phi < 0.0 ) phi += TMath::TwoPi();
415 if( phi > TMath::TwoPi() ) phi -= TMath::TwoPi();
416
417 // Check if we are on side A or B
418 bool sideB = false;
419 if( phi > TMath::TwoPi() / 6. ) {
420
421 sideB = true;
422 phi -= TMath::TwoPi() / 6.;
423
424 }
425
426 // Loop over each strip
427 for( unsigned char i = 0; i < 11; i++ ){
428
429 // Vector for the strip edge
430 TVector2 vec_low( array_radius, ( (double)i-5.5 ) * nstrip_pitch );
431 TVector2 vec_upp( array_radius, ( (double)i-4.5 ) * nstrip_pitch );
432
433 // Rotate by pi/6 to get it aligned with reference, i.e. +30˚
434 // this puts the edge of the wafer in the vertical position
435 vec_low = vec_low.Rotate( TMath::Pi() / 6. );
436 vec_upp = vec_upp.Rotate( TMath::Pi() / 6. );
437
438 // check if we are in the strip
439 if( phi > vec_low.Phi() && phi < vec_upp.Phi() )
440 return i + 11 * sideB;
441
442 }
443
444 // everything else won't hit the silicon
445 return -1;
446
447}
