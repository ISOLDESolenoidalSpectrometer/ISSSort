1140
1141 // Work out the x,y,z position (approximatley)
1142 // We would need to work with coincidences to do this properly
1143 // Then we could avoid some of this and use the ISSEvts class
1144 float z;
1145 int row = asic;
1146 int side = mod * 2 + 1;
1147 if( asic >= 1 ) row--;
1148 if( asic >= 4 ) row--;
1149 if( asic == 0 || asic == 2 || asic == 3 || asic == 5 ){
1150
1151 z = 127.5 - (float)chan; // take centre of the end strip
1152
1153 }
1154 else {
1155
1156 z = 64.0;
1157 if( chan > 64 ) row++;
1158 if( chan < 24 || chan > 104 ) side--;
1159
1160 }
1161
1162 z *= 0.953; // p-side strip pitch = 0.953 mm
1163 z += 1.508; // distance from wafer edge to active region
1164 z += 125.5 * (3.0 - (float)row); // move to correct row (125.0 mm wafer length + 0.5 mm inter-wafer gap)
1165
1166 // Assume it's in the centre of the side
1167 TVector2 vec2( 27.0, 0.0 ); // 54.0 mm face-to-face
1168 vec2 = vec2.Rotate( -1.0 * TMath::Pi() / 6. ); // first face is at -30˚
1169 vec2 = vec2.Rotate( side * 2. * TMath::Pi() / 6. ); // rotate 60˚ for each face

1176
1177 // First calculate the energy loss through the dead layer
1178 if(my_missing_peak_is_last[mod][asic][chan] == true && NumberOfFoundAlphaPeaks < 4)
1179 DetectedEnergy[i] = react->SimulateDecay( vec3, FF_alpha_peak_energy[i] ); // Read the first 3 alpha peak energies

1181 DetectedEnergy[i] = react->SimulateDecay( vec3, FF_alpha_peak_energy[i+1] ); // Read the last 3 alpha peak energies
1182 else
1183 DetectedEnergy[i] = react->SimulateDecay( vec3, FF_alpha_peak_energy[i] ); // Read all 4 alpha peak energies
1184 // Now correct for the pulse-height correction
1185 DetectedEnergy[i] = react->GetPulseHeightCorrection( DetectedEnergy[i], false );
1186
1187 }
1188
1189 // Now do a graph of the centroids against detected energy
1190 TGraphErrors* g = new TGraphErrors( NumberOfFoundAlphaPeaks,
