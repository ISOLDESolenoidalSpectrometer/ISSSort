44// FIT SHAPE 0: GAUSSIAN FIT //
46inline double AlphaGaussian( double *x, double *p ){
47
48 // p[0] sigma
49 // p[1] amp
50 // p[2] mean
51
52 //static float sqrt2pi = TMath::Sqrt( 2.0 * TMath::Pi() );
53 static float sqrt2 = TMath::Sqrt( 2.0 );
54
55 double arg = ( x[0] - p[2] ) / ( sqrt2 * p[0] );
56 double gaus = p[1] * TMath::Exp( -1.0*TMath::Power( arg, 2.0 ) );
57
58 return gaus;
59
60}

63
64 // p[0] gauss constant

69 return AlphaGaussian(x,&p[1]) + p[0];

73inline double MultiAlphaGaussianBG( double *x, double *p ) {
74
75 // par[0] background constant
76 // par[1] gauss width
77 // par[2,4,6,...] gauss constant
78 // par[3,5,7,...] gauss mean
79
80 double bg = p[0];
81 double gaus_par[3];
82
83 double sum = 0;
84 for( int i = 0; i < FF_num_alpha_peaks; i++ ){
85
86 gaus_par[0] = p[1];
87 gaus_par[1] = p[2*i+2];
88 gaus_par[2] = p[2*i+3];
89 sum += AlphaGaussian( x, gaus_par );
90
91 }
92
93 return sum + bg;
94
95}

100inline double CrystalBallFunction( double x, double sigma, double alpha, double n, double amp, double mean ) {
101
102 // Check sigma, n, alpha, and amp have good values
103 if ( sigma <= 0.0 || alpha <= 0.0 || n <= 0.0 || amp <= 0.0){
104 return 0.0;
105 }
106
107 // Define z and absalpha
108 double z = (x - mean)/sigma;
109 double abs_alpha = TMath::Abs( alpha );
110
111 // Define A,B,C,D,N
112 double A = TMath::Power( n/abs_alpha, n)*TMath::Exp( -0.5*TMath::Power( abs_alpha, 2 ) );
113 double B = ( n/abs_alpha ) - abs_alpha;
114
115 // Wikipedia defines C & D, but they're not needed here!
116 //double C = ( n/( abs_alpha*( n - 1 ) ) )*TMath::Exp( -0.5*TMath::Power( abs_alpha, 2 ) );
117 //double D = TMath::Sqrt( TMath::Pi()/2 )*( 1 + TMath::Erf( abs_alpha/TMath::Sqrt(2) ) );
118
119 // Define function
120 if ( z > -abs_alpha ){
121 return amp*TMath::Exp( -0.5*z*z );
122 }
123 else {
124 return amp*A*TMath::Power( B - z, -n );
125 }
126
127}
