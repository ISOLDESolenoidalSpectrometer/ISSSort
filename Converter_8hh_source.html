106 // enumeration for swapping modes
107 enum swap_t {
108 SWAP_KNOWN = 1, // We know the swapping mode already
109 SWAP_WORDS = 2, // We need to swap pairs of 32-bit words
110 SWAP_ENDIAN = 4 // We need to swap endianness
111 };
112 Int_t swap;
113
114 // Swap endianness of a 32-bit integer 0x01234567 -> 0x67452301
115 inline UInt_t Swap32(UInt_t datum) {
116 return(((datum & 0xFF000000) >> 24) |
117 ((datum & 0x00FF0000) >> 8) |
118 ((datum & 0x0000FF00) << 8) |
119 ((datum & 0x000000FF) << 24));
120 };
121
122 // Swap the two halves of a 64-bit integer 0x0123456789ABCDEF ->
123 // 0x89ABCDEF01234567
124 inline ULong64_t SwapWords(ULong64_t datum) {
125 return(((datum & 0xFFFFFFFF00000000LL) >> 32) |
126 ((datum & 0x00000000FFFFFFFFLL) << 32));
127 };
128
129 // Swap endianness of a 64-bit integer 0x0123456789ABCDEF ->
130 // 0xEFCDAB8967452301
131 inline ULong64_t Swap64(ULong64_t datum) {
132 return(((datum & 0xFF00000000000000LL) >> 56) |
133 ((datum & 0x00FF000000000000LL) >> 40) |
134 ((datum & 0x0000FF0000000000LL) >> 24) |
135 ((datum & 0x000000FF00000000LL) >> 8) |
136 ((datum & 0x00000000FF000000LL) << 8) |
137 ((datum & 0x0000000000FF0000LL) << 24) |
138 ((datum & 0x000000000000FF00LL) << 40) |
139 ((datum & 0x00000000000000FFLL) << 56));
140 };
141
142 // Get nth word
143 inline ULong64_t GetWord( UInt_t n = 0 ){
144
145 // If word number is out of range, return zero
146 if( n >= WORD_SIZE ) return(0);
147
148 // Perform byte swapping according to swap mode
149 ULong64_t result = data[n];
150 if (swap & SWAP_ENDIAN) result = Swap64(result);
151 if (swap & SWAP_WORDS) result = SwapWords(result);
152 return(result);
153
154 };
