115 // enumeration for swapping modes
116 enum swap_t {
117 SWAP_KNOWN = 1, // We know the swapping mode already
118 SWAP_WORDS = 2, // We need to swap pairs of 32-bit words
119 SWAP_ENDIAN = 4 // We need to swap endianness
120 };
121 Int_t swap;
122
123 // Swap endianness of a 32-bit integer 0x01234567 -> 0x67452301
124 inline UInt_t Swap32(UInt_t datum) {
125 return(((datum & 0xFF000000) >> 24) |
126 ((datum & 0x00FF0000) >> 8) |
127 ((datum & 0x0000FF00) << 8) |
128 ((datum & 0x000000FF) << 24));
129 };
130
131 // Swap the two halves of a 64-bit integer 0x0123456789ABCDEF ->
132 // 0x89ABCDEF01234567
133 inline ULong64_t SwapWords(ULong64_t datum) {
134 return(((datum & 0xFFFFFFFF00000000LL) >> 32) |
135 ((datum & 0x00000000FFFFFFFFLL) << 32));
136 };
137
138 // Swap endianness of a 64-bit integer 0x0123456789ABCDEF ->
139 // 0xEFCDAB8967452301
140 inline ULong64_t Swap64(ULong64_t datum) {
141 return(((datum & 0xFF00000000000000LL) >> 56) |
142 ((datum & 0x00FF000000000000LL) >> 40) |
143 ((datum & 0x0000FF0000000000LL) >> 24) |
144 ((datum & 0x000000FF00000000LL) >> 8) |
145 ((datum & 0x00000000FF000000LL) << 8) |
146 ((datum & 0x0000000000FF0000LL) << 24) |
147 ((datum & 0x000000000000FF00LL) << 40) |
148 ((datum & 0x00000000000000FFLL) << 56));
149 };
150
151 // Get nth word
152 inline ULong64_t GetWord( UInt_t n = 0 ){
153
154 // If word number is out of range, return zero
155 if( n >= WORD_SIZE ) return(0);
156
157 // Perform byte swapping according to swap mode
158 ULong64_t result = data[n];
159 if (swap & SWAP_ENDIAN) result = Swap64(result);
160 if (swap & SWAP_WORDS) result = SwapWords(result);
161 return(result);
162
163 };
