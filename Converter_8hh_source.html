100 // enumeration for swapping modes
101 enum swap_t {
102 SWAP_KNOWN = 1, // We know the swapping mode already
103 SWAP_WORDS = 2, // We need to swap pairs of 32-bit words
104 SWAP_ENDIAN = 4 // We need to swap endianness
105 };
106 Int_t swap;
107
108 // Swap endianness of a 32-bit integer 0x01234567 -> 0x67452301
109 inline UInt_t Swap32(UInt_t datum) {
110 return(((datum & 0xFF000000) >> 24) |
111 ((datum & 0x00FF0000) >> 8) |
112 ((datum & 0x0000FF00) << 8) |
113 ((datum & 0x000000FF) << 24));
114 };
115
116 // Swap the two halves of a 64-bit integer 0x0123456789ABCDEF ->
117 // 0x89ABCDEF01234567
118 inline ULong64_t SwapWords(ULong64_t datum) {
119 return(((datum & 0xFFFFFFFF00000000LL) >> 32) |
120 ((datum & 0x00000000FFFFFFFFLL) << 32));
121 };
122
123 // Swap endianness of a 64-bit integer 0x0123456789ABCDEF ->
124 // 0xEFCDAB8967452301
125 inline ULong64_t Swap64(ULong64_t datum) {
126 return(((datum & 0xFF00000000000000LL) >> 56) |
127 ((datum & 0x00FF000000000000LL) >> 40) |
128 ((datum & 0x0000FF0000000000LL) >> 24) |
129 ((datum & 0x000000FF00000000LL) >> 8) |
130 ((datum & 0x00000000FF000000LL) << 8) |
131 ((datum & 0x0000000000FF0000LL) << 24) |
132 ((datum & 0x000000000000FF00LL) << 40) |
133 ((datum & 0x00000000000000FFLL) << 56));
134 };
135
136 // Get nth word
137 inline ULong64_t GetWord( UInt_t n = 0 ){
138
139 // If word number is out of range, return zero
140 if( n >= WORD_SIZE ) return(0);
141
142 // Perform byte swapping according to swap mode
143 ULong64_t result = data[n];
144 if (swap & SWAP_ENDIAN) result = Swap64(result);
145 if (swap & SWAP_WORDS) result = SwapWords(result);
146 return(result);
147
148 };
