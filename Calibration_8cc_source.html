11double walk_function( double *x, double *params ){
12
13 double deltaT = x[0];
14 double A = params[0];
15 double B = params[1];
16 double C = params[2];
17 double D = params[3];
18 double E = params[4];
19
20 // Root to solve
21 double root = A;
22 root += B * deltaT;
23 root += TMath::Exp( C + D * deltaT );
24 root -= E;
25
26 return root;

433float ISSCalibration::AsicWalk( unsigned int mod, unsigned int asic, float energy, bool hit ){
434
435 float walk = 0;
436
437 if( mod < set->GetNumberOfArrayModules() &&
438 asic < set->GetNumberOfArrayASICs() ) {
439
440
441 if( fAsicWalkType[mod][asic] == 2 ) {
442
443 if( hit )walk = -tw_graph[mod][asic][1]->Eval(energy);
444 else walk = -tw_graph[mod][asic][0]->Eval(energy);
445
446 }
447
448 else{
449 // Check if all values are defaulted to zero - no walk correction
450 bool nowalk = true;
451 for( unsigned int i = 0; i < nwalkpars; i++ ) {
452
453 if( ( TMath::Abs( fAsicWalkHit0[mod][asic][i] ) > 1.0e-6 && !hit ) ||
454 ( TMath::Abs( fAsicWalkHit1[mod][asic][i] ) > 1.0e-6 && hit ) )
455 nowalk = false;
456
457 }
458
459 // If no walk correction, just return 0
460 if( nowalk ) return 0.0;
461
462 // else calculate the walk using the defined function
463 else {
464
465 // Params for time walk function ROOT finder
466 for( unsigned int i = 0; i < nwalkpars; i++ ){
467
468 if( hit ) walk_params[i] = fAsicWalkHit1[mod][asic][i];
469 else walk_params[i] = fAsicWalkHit0[mod][asic][i];
470
471 }
472
473 // Last one is always the energy
474 walk_params[nwalkpars] = energy;
475
476 // Annie Dolan's function
477 if( fAsicWalkType[mod][asic] == 0 ) {
478
479 // Set parameters
480 fa->SetParameters( walk_params );
481 fb->SetParameters( walk_params );
482
483 // Build the function and derivative, then solve
484 gErrorIgnoreLevel = kBreak; // suppress warnings and errors, but not breaks
485 ROOT::Math::GradFunctor1D wf( *fa, *fb );
486 rf->SetFunction( wf, -2e4, 2e4 ); // limits
487 rf->Solve( 500, 1e-4, 1e-5 );
488
489 // Check result
490 if( rf->Status() ){
491 walk = TMath::QuietNaN();
492 }
493 else walk = rf->Root();
494 gErrorIgnoreLevel = kInfo; // print info and above again
495
496 }
497
498 // Sam Reeve's function
499 if( fAsicWalkType[mod][asic] == 1 ) {
500
501 // Functional form: y = a + b / (c*E)^d
502 // where y is time walk and E is energy
503 walk = walk_params[1];
504 walk /= TMath::Power( walk_params[2]*walk_params[4], walk_params[3] );
505 walk += walk_params[0];
506
507
508 // Find limit inside event window
509 // Solved for time walk = 3000 ns
510 // E = ( b / (y - a) )^(1/d) / c
511 double E_limit;
512 double dT_limit = 3000;
513 E_limit = walk_params[1];
514 E_limit /= dT_limit - walk_params[0];
515 E_limit = TMath::Power( E_limit, 1/walk_params[3] );
516 E_limit /= walk_params[2];
517 E_limit = TMath::Abs(E_limit);
518
519 // Add minus as the correction is added to time in EventBuilder
520 if( walk_params[4] < E_limit ){
521 walk = -3000;
522 }
523 else{
524 walk = -walk;
525 }
526
527 }
528
529 }
530
531 }
532
533 }
534
535
536
537 return walk;
538
